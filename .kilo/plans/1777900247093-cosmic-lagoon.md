### **Arquitetura do Sistema**

#### **1. Estrutura de Dados (PrintData & PrintOptions)**
```typescript
interface PrintData {
  order: Order;
  client: Client;
  store?: Store;
  lab?: Lab;
  installments?: Installment[];
  qrCodeUrl?: string;
  generatedAt: Date;
  printedBy?: string;
}

interface PrintOptions {
  templateId: string;
  format: 'pdf' | 'png' | 'jpg';
  includeQR: boolean;
  paperSize: 'a4' | 'a5' | 'letter';
  orientation: 'portrait' | 'landscape';
  copies: number;
}
```

#### **2. Templates Disponíveis**
- **OrderReceiptTemplate**: Recibo padrão do pedido
- **OrderLabelTemplate**: Etiqueta para identificação
- **OrderSummaryTemplate**: Resumo do pedido
- **CarnePagamentoTemplate**: Carnê de pagamentos
- **OrderPDF**: Template especial para envelope/ficha

#### **3. Fluxo de Geração**
```
Cliente solicita PDF → API busca dados → Gera componente React → 
renderToBuffer() → Buffer → Blob → Download/Preview
```

### **Implementação Técnica Detalhada**

#### **Geração de PDF (pdfGenerator.ts)**
```typescript
// Função principal que converte componente React em buffer PDF
export async function generatePDFBuffer(
  data: PrintData,
  templateId: string,
  options: PrintOptions
): Promise<Buffer> {
  let element: React.ReactElement;
  
  if (templateId === 'envelope' || templateId === 'ficha') {
    element = React.createElement(OrderPDF, { ...props });
  } else {
    const TemplateComponent = await getTemplateComponent(templateId);
    element = React.createElement(TemplateComponent, { data, options });
  }

  const pdfBuffer = await renderToBuffer(element as any);
  return pdfBuffer;
}
```

#### **API Endpoint (generate-pdf/route.ts)**
```typescript
export async function POST(request: NextRequest) {
  const { orderId, templateId, options = {} } = await request.json();
  
  // Busca dados completos do pedido
  const printData = await getOrderPrintData(orderId);
  
  // Gera PDF
  const pdfBuffer = await generatePDFBuffer(printData, templateId, printOptions);
  
  // Retorna como resposta HTTP
  return new NextResponse(new Uint8Array(pdfBuffer), {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename="pedido-${numeroPedido}.pdf"`
    }
  });
}
```

import { generatePDFBuffer } from '@/lib/printing/pdfGenerator';

export async function POST(request: Request) {
  const { data, templateId } = await request.json();
  
  const buffer = await generatePDFBuffer(data, templateId, {});
  
  return new Response(buffer, {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': 'attachment; filename="document.pdf"'
    }

}
